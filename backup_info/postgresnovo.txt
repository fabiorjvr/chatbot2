                        """, (row['fabricante'], 'Diversos', 0, f'Fabricante {row["fabricante"]}'))
                        fabricante_id = cur.fetchone()[0]
                    else:
                        fabricante_id = fabricante_result[0]
                    
                    # Atualizar ou inserir produto
                    cur.execute("""
                        INSERT INTO produtos

    def import_inventario(self, csv_path: str):

        logger.info("üì¶ IMPORTANDO INVENT√ÅRIO - PHONE PARAGUAY...")
